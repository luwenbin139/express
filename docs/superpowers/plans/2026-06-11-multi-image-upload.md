# Multi Image Upload Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add multi-image upload support to the existing Express image APIs and replace the missing frontend source with a maintainable frontend that builds into root `build/`.

**Architecture:** Keep `index.js` as the Express entrypoint, but extract request parsing into focused helpers so JSON and multipart requests share the same validation path. Add a new `frontend/` Vite React app that posts `multipart/form-data` to the existing API and builds directly to `../build`.

**Tech Stack:** Node.js CommonJS, Express 4, multer for multipart parsing, plain Node assertion tests, Vite + React + TypeScript frontend.

**Git note:** Do not run `git commit` unless the user explicitly asks. Each task lists files to stage/review instead of commit commands.

function collectJsonImageDataUrls(body) {
  const imageCandidates = [];

  if (body && typeof body.image === "string") {
    imageCandidates.push(body.image);
  }

  if (body && Array.isArray(body.images)) {
    for (const image of body.images) {
      if (typeof image === "string") {

  return imageCandidates.map((image) => image.trim()).filter(Boolean);

      setMessage('еӣҫзүҮзј–иҫ‘жЁЎејҸйңҖиҰҒиҮіе°‘дёҠдј  1 еј еҺҹеӣҫ');
      return;
    }

    const formData = new FormData();
    formData.append('prompt', trimmedPrompt);
    formData.append('mode', mode);
    formData.append('size', size);

    for (const file of files) {
      formData.append('images', file);
    }

    setLoading(true);
    setMessage('жӯЈеңЁжҸҗдәӨпјҢиҜ·зЁҚеҖҷ...');
    setResultImage('');

    try {
      const response = await fetch('/api/generate-image-stream', {
        method: 'POST',
        body: formData,
      });
      const data = (await response.json()) as ApiResponse;

      if (!response.ok || data.code !== 0 || !data.data?.image) {
        throw new Error(data.message || 'еӣҫзүҮз”ҹжҲҗеӨұиҙҘ');
      }

      setResultImage(data.data.image);
      setMessage(`з”ҹжҲҗе®ҢжҲҗпјҢжҺҘеҸЈпјҡ${data.data.api}`);
    } catch (error) {
      setMessage(error instanceof Error ? error.message : 'еӣҫзүҮз”ҹжҲҗеӨұиҙҘ');
    } finally {
      setLoading(false);
    }
  }

  return (
    <main className="page">
      <section className="panel">
        <div className="header">
          <p className="eyebrow">AI Image Studio</p>
          <h1>еӨҡеӣҫзүҮдёҠдј з”ҹжҲҗ</h1>
          <p>ж”ҜжҢҒз”ҹжҲҗж–°еӣҫпјҢд№ҹж”ҜжҢҒдёҠдј жңҖеӨҡ 4 еј еҸӮиҖғеӣҫиҝӣиЎҢзј–иҫ‘гҖӮ</p>
        </div>

        <form className="form" onSubmit={submit}>
          <label>
            жҸҗзӨәиҜҚ
            <textarea
              value={prompt}
              onChange={(event) => setPrompt(event.target.value)}
              maxLength={2000}
              placeholder="жҸҸиҝ°дҪ жғіз”ҹжҲҗжҲ–зј–иҫ‘зҡ„еӣҫзүҮ..."
            />
          </label>

          <div className="grid">
            <label>
              жЁЎејҸ
              <select value={mode} onChange={(event) => setMode(event.target.value as 'generate' | 'edit')}>
                <option value="generate">з”ҹжҲҗж–°еӣҫ</option>
                <option value="edit">еҸӮиҖғеӣҫзј–иҫ‘</option>
              </select>
            </label>

            <label>
              е°әеҜё
              <select value={size} onChange={(event) => setSize(event.target.value)}>
                <option value="1024x1024">1024 Г— 1024</option>
                <option value="1024x1536">1024 Г— 1536</option>
                <option value="1536x1024">1536 Г— 1024</option>
                <option value="1920x1080">1920 Г— 1080</option>
              </select>
            </label>
          </div>

          <label className="upload">
            <span>дёҠдј еҸӮиҖғеӣҫпјҢжңҖеӨҡ 4 еј пјҢжҜҸеј  10MB</span>
            <input
              type="file"
              accept="image/png,image/jpeg,image/webp"
              multiple
              onChange={(event) => onPickFiles(Array.from(event.target.files || []))}
            />
          </label>

          {previews.length > 0 && (
            <div className="previews">
              {previews.map(({ file, url }, index) => (
                <figure key={`${file.name}-${index}`}>
                  <img src={url} alt={file.name} />
                  <figcaption>{file.name}</figcaption>
                  <button type="button" onClick={() => removeFile(index)}>з§»йҷӨ</button>
                </figure>
              ))}
            </div>
          )}

          <button className="submit" type="submit" disabled={loading}>
            {loading ? 'з”ҹжҲҗдёӯ...' : 'ејҖе§Ӣз”ҹжҲҗ'}
          </button>
        </form>

        {message && <p className="message">{message}</p>}

        {resultImage && (
          <section className="result">
            <h2>з”ҹжҲҗз»“жһң</h2>
            <img src={resultImage} alt="з”ҹжҲҗз»“жһң" />
          </section>
        )}
      </section>
    </main>
  );
}
